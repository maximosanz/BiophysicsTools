def Setup_Plot(X,Y,Yfit,fs=12,col='k'):

    plt.scatter(X,Y,s=4,color=col)
    plt.xlabel("Temperature (C)",fontsize=fs)
    plt.ylabel("Molar ellipticity at 222 nm",fontsize=fs)
    return
